# Agglomerations Specification (T_Agglomerations)

**DISCO Table**: `[WISE_UWWTD].[v1r1].[T_Agglomerations]`  
**Description**: Population agglomerations that generate urban waste water  
**Record Count**: ~25,000

## Upload Logic

1. **Extract**: Download from DISCO API
2. **Validate**:
   - Check `aggCode` uniqueness
   - Validate country codes
   - Verify C1, C2 percentages are 0-100
   - Check lat/lon ranges
3. **Transform**:
   - Map DISCO column names to database columns
   - Parse percentage values
   - Convert dates
4. **Load**: Insert with upsert logic

## Judgment Logic

### Article 3 Compliance (Collection System)
```python
def check_article3_compliance(agg_row):
    """
    Check Article 3 compliance - collecting system requirements.
    Agglomerations >= 2000 PE must have collecting system.
    """
    agg_generated = agg_row.get('aggGenerated', 0)
    agg_c1 = agg_row.get('aggC1', 0)  # % connected to collecting system
    
    if agg_generated < 2000:
        return {
            'required': False,
            'compliant': True,
            'reason': 'Below 2000 PE threshold'
        }
    
    # >= 2000 PE requires collecting system
    if agg_c1 >= 100:
        return {
            'required': True,
            'compliant': True,
            'reason': f'100% connected (C1={agg_c1}%)'
        }
    else:
        return {
            'required': True,
            'compliant': False,
            'reason': f'Only {agg_c1}% connected (C1), 100% required'
        }
```

### Article 4 Compliance (Secondary Treatment)
```python
def check_article4_compliance(agg_row, linked_plants):
    """
    Check Article 4 compliance - secondary treatment requirements.
    Agglomerations >= 2000 PE must have secondary treatment.
    """
    agg_generated = agg_row.get('aggGenerated', 0)
    agg_c2 = agg_row.get('aggC2', 0)  # % receiving appropriate treatment
    
    if agg_generated < 2000:
        return {
            'required': False,
            'compliant': True,
            'reason': 'Below 2000 PE threshold'
        }
    
    # Check if linked plants provide secondary treatment
    has_secondary = any(p.get('uwwSecondaryTreatment') for p in linked_plants)
    
    if agg_c2 >= 100 and has_secondary:
        return {
            'required': True,
            'compliant': True,
            'reason': f'100% treated (C2={agg_c2}%) with secondary treatment'
        }
    else:
        issues = []
        if agg_c2 < 100:
            issues.append(f'Only {agg_c2}% receiving treatment')
        if not has_secondary:
            issues.append('No linked plant with secondary treatment')
        return {
            'required': True,
            'compliant': False,
            'reason': '; '.join(issues)
        }
```

### Article 5 Compliance (Nutrient Removal)
```python
def check_article5_compliance(agg_row, linked_plants, discharge_points):
    """
    Check Article 5 compliance - nutrient removal in sensitive areas.
    """
    agg_generated = agg_row.get('aggGenerated', 0)
    
    # Check if any discharge point is in sensitive area
    sensitive_codes = ['A54', 'A523', 'A5854', 'A58523', 'CSA']
    is_sensitive = any(
        dp.get('dcpTypeOfReceivingArea') in sensitive_codes 
        for dp in discharge_points
    )
    
    if not is_sensitive:
        return {
            'required': False,
            'compliant': True,
            'reason': 'Not discharging to sensitive area'
        }
    
    result = {
        'required': True,
        'p_removal_required': agg_generated >= 10000,
        'n_removal_required': agg_generated >= 100000,
        'issues': []
    }
    
    # Check phosphorus removal
    if result['p_removal_required']:
        has_p_removal = any(p.get('uwwPRemoval') for p in linked_plants)
        if not has_p_removal:
            result['issues'].append('P removal required but not provided')
    
    # Check nitrogen removal
    if result['n_removal_required']:
        has_n_removal = any(p.get('uwwNRemoval') for p in linked_plants)
        if not has_n_removal:
            result['issues'].append('N removal required but not provided')
    
    result['compliant'] = len(result['issues']) == 0
    result['reason'] = '; '.join(result['issues']) if result['issues'] else 'Compliant'
    
    return result
```

### Treatment Tier Assignment
```python
def assign_treatment_tier(agg_generated):
    """
    Assign treatment tier based on agglomeration generated load.
    """
    if agg_generated < 1000:
        return {
            'tier': 'Individual/Small',
            'code': 0,
            'requirements': 'Appropriate treatment'
        }
    elif agg_generated < 2000:
        return {
            'tier': 'Tier 1 (1000-1999 PE)',
            'code': 1,
            'requirements': 'Secondary treatment (with derogations possible)'
        }
    elif agg_generated < 10000:
        return {
            'tier': 'Tier 2 (2000-9999 PE)',
            'code': 2,
            'requirements': 'Secondary treatment'
        }
    elif agg_generated < 100000:
        return {
            'tier': 'Tier 3 (10000-99999 PE)',
            'code': 3,
            'requirements': 'Secondary treatment + P removal in sensitive areas'
        }
    else:
        return {
            'tier': 'Tier 4 (>= 100000 PE)',
            'code': 4,
            'requirements': 'Secondary treatment + N and P removal in sensitive areas'
        }
```

### Collection System Coverage Analysis
```python
def analyze_collection_coverage(agg_row):
    """
    Analyze collection system coverage and treatment distribution.
    """
    c1 = agg_row.get('aggC1', 0)  # % connected to collecting system
    c2 = agg_row.get('aggC2', 0)  # % receiving appropriate treatment
    
    without_treatment = agg_row.get('aggPercWithoutTreatment', 0)
    primary = agg_row.get('aggPercPrimTreatment', 0)
    secondary = agg_row.get('aggPercSecTreatment', 0)
    stringent = agg_row.get('aggPercStringentTreatment', 0)
    
    return {
        'collection_rate': c1,
        'treatment_rate': c2,
        'gap': c1 - c2,  # Connected but not treated
        'treatment_distribution': {
            'without_treatment': without_treatment,
            'primary_only': primary,
            'secondary': secondary,
            'stringent': stringent
        },
        'quality_issues': c1 < 100 or c2 < 100
    }
```

### Data Quality Assessment
```python
def assess_agglomeration_data_quality(agg_row):
    """
    Assess data quality and completeness for an agglomeration record.
    """
    issues = []
    
    # Required fields
    if not agg_row.get('aggCode'):
        issues.append('Missing agglomeration code')
    
    if not agg_row.get('aggName'):
        issues.append('Missing agglomeration name')
    
    # Location data
    lat, lon = agg_row.get('aggLatitude'), agg_row.get('aggLongitude')
    if not lat or not lon:
        issues.append('Missing coordinates')
    
    # Load data
    if not agg_row.get('aggGenerated'):
        issues.append('Missing generated load')
    
    # Coverage data
    if agg_row.get('aggC1') is None:
        issues.append('Missing C1 (collection coverage)')
    
    if agg_row.get('aggC2') is None:
        issues.append('Missing C2 (treatment coverage)')
    
    # Consistency check
    c1 = agg_row.get('aggC1', 0)
    c2 = agg_row.get('aggC2', 0)
    if c2 > c1:
        issues.append('C2 > C1 (treatment rate exceeds collection rate)')
    
    return {
        'complete': len(issues) == 0,
        'issues': issues,
        'completeness_score': max(0, 100 - len(issues) * 15)
    }
```
